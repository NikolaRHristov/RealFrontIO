// Inject into multiLensFrag from Batch 10

const multiLensFrag = `
precision highp float;

varying vec2 vTextureCoord;
uniform sampler2D uSampler;

uniform sampler2D uDataBlockA; // [0-3]: Seismic, Fire(Thermal), Cyber, GPS
uniform sampler2D uDataBlockB; 
uniform sampler2D uDataBlockC; 

// A separate scalar texture representing the Sun's current illumination over the grid
uniform sampler2D uSolarIllumination; 

uniform int uActiveLens; 

void main(void) {
    vec4 baseColor = texture2D(uSampler, vTextureCoord);
    
    vec4 blockA = texture2D(uDataBlockA, vTextureCoord);
    float sunlight = texture2D(uSolarIllumination, vTextureCoord).r; // 0.0 = Night, 1.0 = Day
    
    // Default Visuals (Visual Light Spectrum)
    if (uActiveLens == -1) {
        // Multiply the base texture by the sun's illumination.
        // If it is 2 AM in Tokyo in real life, Tokyo is pitch black in-game.
        gl_FragColor = vec4(baseColor.rgb * sunlight, baseColor.a);
        return;
    }

    float severity = 0.0;
    vec3 brutalColor = vec3(0.0);

    // --- THERMAL LENS (Reads Channel 1 / Fire) ---
    if (uActiveLens == 4) {
        // In thermal, ambient light doesn't matter. We only care about heat signatures.
        float heat = blockA.g; // Channel 1 (Fire/Thermal)
        
        // Cold background (Blue/Purple)
        vec3 coldBg = vec3(0.1, 0.0, 0.3);
        
        // White-Hot objects (Units, Cities, NASA Wildfires)
        vec3 hotObj = vec3(1.0, 1.0, 1.0);
        
        // Mid-tier heat (Red/Orange)
        vec3 midHeat = vec3(1.0, 0.3, 0.0);

        if (heat < 0.3) {
            brutalColor = mix(coldBg, midHeat, heat / 0.3);
        } else {
            brutalColor = mix(midHeat, hotObj, (heat - 0.3) / 0.7);
        }
        
        // Add extreme grain/noise to simulate FLIR cameras
        float flirNoise = fract(sin(dot(vTextureCoord * 12.9898, vec2(78.233, 45.164))) * 43758.5453);
        brutalColor += (flirNoise * 0.1);
        
        gl_FragColor = vec4(brutalColor, 1.0);
        return;
    }
    
    // ... rest of the brutalist lenses from Batch 10
}
`;
